bem_show_decisions_gbr_toggle = {
	is_shown = {
		exists = c:GBR
		this = c:GBR
		NOT = {
			has_variable = show_bem_decisions_var
		}
		is_player = yes
	}

	possible = {
		always = yes
	}

	when_taken = {
		set_variable = show_bem_decisions_var
	}
	
	ai_chance = {
		base = 0 #AI should never do this
	}
}
